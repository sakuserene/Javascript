# 1. The Problem: Mixed Code

## Bad Example

```js id="8blt1x"
function addProduct() {

  const name = input.value;

  const product = {
    id: Date.now(),
    name: name
  };

  products.push(product);

  const li = document.createElement("li");
  li.textContent = product.name;

  list.appendChild(li);
}
```

---

## Problems

* Updates state ❌
* Updates DOM ❌
* Handles input ❌
* Too many responsibilities ❌

# 2. Single Responsibility Principle

Each function should do ONE thing.


# 3. Clean Architecture Pattern

## Structure

```text id="vljptm"

```

# 4. Refactoring the Example

## Step 1: State Function

```js id="4ys9rd"
function addProductToState(data) {
  const newProduct = {
    id: Date.now(),
    ...data
  };

  products = [...products, newProduct];

  return newProduct;
}
```

---

## Step 2: Render Function

```js id="h3c7m1"
function renderProducts() {
  list.innerHTML = "";

  products.forEach(product => {
    const li = document.createElement("li");
    li.textContent = product.name;
    list.appendChild(li);
  });
}
```

---

## Step 3: Event Handler

```js id="8vb3g7"
form.addEventListener("submit", function(event) {
  event.preventDefault();

  const data = {
    name: input.value
  };

  addProductToState(data);

  renderProducts();
});
```

# 5. Three Complete Examples

## Example 1: Basic Separation

```js id="7k7h5k"
let numbers = [];

function addNumber(n) {
  numbers.push(n);
}

function renderNumbers() {
  console.log(numbers);
}
```

---

## Example 2: Object State

```js id="tfj6p2"
function updateProduct(id, newName) {
  products = products.map(product =>
    product.id === id
      ? { ...product, name: newName }
      : product
  );
}
```

---

## Example 3: Full DOM Application

```js id="bnspmt"
form.addEventListener("submit", function(event) {
  event.preventDefault();

  const data = {
    name: nameInput.value
  };

  const error = validateProduct(data);

  if (error) {
    showError(error);
    return;
  }

  addProductToState(data);

  saveProducts();
  renderProducts();
});
```

# 7. Comparing Bad vs Good

## Bad

```js id="0d1cl7"
function deleteProduct(id) {
  // removes from state
  // removes from DOM
  // logs
}
```

---

## Good

```js id="mg42mz"
function deleteProductFromState(id) {
  products = products.filter(p => p.id !== id);
}

function renderProducts() {
  ...
}
```
