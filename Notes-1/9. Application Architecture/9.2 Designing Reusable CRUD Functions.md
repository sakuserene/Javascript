# 1. The Problem: Repeated Code

## Bad Example

```js
function updateProductName(id, name) {
  products = products.map(p =>
    p.id === id ? { ...p, name } : p
  );
}

function updateProductPrice(id, price) {
  products = products.map(p =>
    p.id === id ? { ...p, price } : p
  );
}
```

---

## Problem

* Duplicate logic ❌
* Hard to maintain ❌
* Not scalable ❌

# 2. Reusable Function Concept

Instead of writing many functions:

```text
One reusable function → many use cases
```

---

# 3. Generic Update Function

## Concept Explanation

We pass:

* ID
* Field to update
* New value

---

## Implementation

```js
function updateItemById(id, updates) {
  products = products.map(item =>
    item.id === id
      ? { ...item, ...updates }
      : item
  );
}
```

---

## Usage

```js
updateItemById(1, { name: "Tablet" });

updateItemById(1, { price: 30000 });
```

---

# 4. Generic Delete Function

```js
function deleteItemById(id) {
  products = products.filter(item => item.id !== id);
}
```

---

# 5. Generic Find Function

```js
function findItemById(id) {
  return products.find(item => item.id === id);
}
```

---

# 6. Generic Create Function

```js
function createItem(data) {
  const newItem = {
    id: Date.now(),
    ...data
  };

  products = [...products, newItem];

  return newItem;
}
```

---

# 7. Three Complete Examples

## Example 1: Basic Numbers

```js
let numbers = [1,2,3];

function deleteNumber(n) {
  numbers = numbers.filter(x => x !== n);
}
```

## Example 2: Object CRUD

# 9. Handling Missing Items

```js
function updateItemById(id, updates) {
  let found = false;

  products = products.map(item => {
    if (item.id === id) {
      found = true;
      return { ...item, ...updates };
    }
    return item;
  });

  if (!found) {
    console.warn("Item not found");
  }
}
```

---

# 10. Making Functions Flexible

## Concept

Pass collection as parameter:

```js
function deleteById(list, id) {
  return list.filter(item => item.id !== id);
}
```

---

## Usage

```js
products = deleteById(products, 1);
tasks = deleteById(tasks, 2);
```
