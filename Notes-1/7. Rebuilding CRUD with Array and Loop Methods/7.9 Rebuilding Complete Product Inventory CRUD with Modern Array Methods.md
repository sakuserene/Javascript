# 12. Duplicate Validation with `some()`

```js
function productNameExists(
  productName,
  ignoredProductId = null
) {
  const normalizedName =
    normalizeProductName(
      productName
    );

  return products.some(
    (product) =>
      product.id !==
        ignoredProductId &&
      normalizeProductName(
        product.name
      ) === normalizedName
  );
}
```

# 13. Form Validation

```js
function validateProductForm(
  data
) {
  if (data.name === "") {
    return {
      isValid: false,
      field: "name",
      message:
        "Product name is required.",
    };
  }

  if (data.name.length < 2) {
    return {
      isValid: false,
      field: "name",
      message:
        "Product name must contain at least 2 characters.",
    };
  }

  if (data.name.length > 80) {
    return {
      isValid: false,
      field: "name",
      message:
        "Product name cannot exceed 80 characters.",
    };
  }

  if (
    productNameExists(
      data.name,
      editingProductId
    )
  ) {
    return {
      isValid: false,
      field: "name",
      message:
        "A product with this name already exists.",
    };
  }

  if (data.priceText === "") {
    return {
      isValid: false,
      field: "price",
      message:
        "Product price is required.",
    };
  }

  const price =
    Number(data.priceText);

  if (
    !Number.isFinite(price)
  ) {
    return {
      isValid: false,
      field: "price",
      message:
        "Product price must be a valid number.",
    };
  }

  if (price < 0) {
    return {
      isValid: false,
      field: "price",
      message:
        "Product price cannot be negative.",
    };
  }

  if (
    data.quantityText === ""
  ) {
    return {
      isValid: false,
      field: "quantity",
      message:
        "Product quantity is required.",
    };
  }

  const quantity =
    Number(
      data.quantityText
    );

  if (
    !Number.isInteger(
      quantity
    )
  ) {
    return {
      isValid: false,
      field: "quantity",
      message:
        "Product quantity must be a whole number.",
    };
  }

  if (quantity < 0) {
    return {
      isValid: false,
      field: "quantity",
      message:
        "Product quantity cannot be negative.",
    };
  }

  return {
    isValid: true,
    field: null,
    message: "",
    normalizedData: {
      name: data.name,
      price,
      quantity,
    },
  };
}
```

# 14. Creating Products

## Mutable Create with `push()`

```js
function createProduct(
  data
) {
  const newProduct = {
    id: nextProductId,
    name: data.name,
    price: data.price,
    quantity: data.quantity,
  };

  nextProductId++;

  products.push(
    newProduct
  );

  return newProduct;
}
```

---

## Immutable Create with Spread

You can instead write:

```js
function createProduct(
  data
) {
  const newProduct = {
    id: nextProductId,
    name: data.name,
    price: data.price,
    quantity: data.quantity,
  };

  nextProductId++;

  products = [
    ...products,
    newProduct,
  ];

  return newProduct;
}
```

This application will use the immutable version.

# 17. Updating Products with `map()`

```js
function updateProductById(
  productId,
  data
) {
  const existingProduct =
    findProductById(
      productId
    );

  if (
    existingProduct ===
    undefined
  ) {
    return null;
  }

  products =
    products.map(
      (product) =>
        product.id === productId
          ? {
              ...product,
              name: data.name,
              price: data.price,
              quantity:
                data.quantity,
            }
          : product
    );

  return findProductById(
    productId
  );
}
```

This creates:

* A new products array.
* A new object for the matching product.
* Existing object references for unchanged products.

# 18. Deleting Products with `filter()`

```js
function deleteProductById(
  productId
) {
  const productToDelete =
    findProductById(
      productId
    );

  if (
    productToDelete ===
    undefined
  ) {
    return null;
  }

  products =
    products.filter(
      (product) =>
        product.id !==
        productId
    );

  return productToDelete;
}
```

This creates a new array containing every product except the selected product.

# 19. Product Validation with `every()`

```js
function isValidProduct(
  product
) {
  return (
    Number.isInteger(
      product.id
    ) &&
    product.id > 0 &&
    typeof product.name ===
      "string" &&
    product.name.trim() !==
      "" &&
    Number.isFinite(
      product.price
    ) &&
    product.price >= 0 &&
    Number.isInteger(
      product.quantity
    ) &&
    product.quantity >= 0
  );
}
```

Check the complete state:

```js
function allProductsAreValid() {
  return products.every(
    isValidProduct
  );
}
```

An empty array returns `true`.

For this application, an empty inventory is allowed.

# 20. Summary Calculation with `reduce()`

```js
function calculateInventorySummary() {
  return products.reduce(
    (
      summary,
      product
    ) => {
      summary.totalUnits +=
        product.quantity;

      summary.inventoryValue +=
        product.price *
        product.quantity;

      if (
        product.quantity === 0
      ) {
        summary.outOfStockCount++;
      }

      return summary;
    },
    {
      totalUnits: 0,
      inventoryValue: 0,
      outOfStockCount: 0,
    }
  );
}
```

# 21. Search and Stock Filtering with `filter()`

```js
function getVisibleProducts() {
  const normalizedSearchTerm =
    normalizeProductName(
      currentSearchTerm
    );

  return products.filter(
    (product) => {
      const matchesSearch =
        normalizeProductName(
          product.name
        ).includes(
          normalizedSearchTerm
        );

      const matchesStockFilter =
        currentStockFilter ===
          "all" ||
        (
          currentStockFilter ===
            "in-stock" &&
          product.quantity > 0
        ) ||
        (
          currentStockFilter ===
            "out-of-stock" &&
          product.quantity === 0
        );

      return (
        matchesSearch &&
        matchesStockFilter
      );
    }
  );
}
```

This function does not replace:

```js
products
```

It returns derived visible data.

```js
function handleProductFormSubmit(
  event
) {
  event.preventDefault();

  clearFeedback();
  clearFieldErrors();

  const rawData =
    readProductForm();

  const validationResult =
    validateProductForm(
      rawData
    );

  if (
    !validationResult.isValid
  ) {
    showFieldError(
      validationResult.field,
      validationResult.message
    );

    return;
  }

  if (
    editingProductId === null
  ) {
    const newProduct =
      createProduct(
        validationResult.normalizedData
      );

    resetFormToCreateMode();
    renderProducts();

    showFeedback(
      `${newProduct.name} was added successfully.`
    );

    return;
  }

  const productId =
    editingProductId;

  const updatedProduct =
    updateProductById(
      productId,
      validationResult.normalizedData
    );

  if (
    updatedProduct === null
  ) {
    resetFormToCreateMode();
    renderProducts();

    showFeedback(
      "The selected product could not be updated because it no longer exists."
    );

    return;
  }

  const updatedName =
    updatedProduct.name;

  resetFormToCreateMode();
  renderProducts();

  showFeedback(
    `${updatedName} was updated successfully.`
  );
}
```

```js
function handleDeleteProduct(
  productId
) {
  clearFeedback();

    findProductById(
      productId
    );

    product === undefined
  ) {

    );

    return false;
  }

  const confirmed =
    window.confirm(
      `Are you sure you want to delete ${product.name}?`
    );

  if (!confirmed) {
    showFeedback(
      `${product.name} was not deleted.`
    );

    return false;
  }

  const deletedProduct =
    deleteProductById(
      productId
    );

  if (
    deletedProduct === null
  ) {
    showFeedback(
      "The product could not be deleted."
    );

    renderProducts();
    return false;
  }

  if (
    editingProductId ===
    deletedProduct.id
  ) {
    resetFormToCreateMode();
  }

  renderProducts();

  showFeedback(
    `${deletedProduct.name} was deleted successfully.`
  );

  return true;
}
```

```js
function handleProductFormSubmit(
  event
) {

  clearFeedback();
  clearFieldErrors();

    renderProducts();

    showFeedback(

  }

  ) {

    `${updatedName} was updated successfully.`
  );
}

function handleDeleteProduct(
  productId
) {
  clearFeedback();

  const product =
    findProductById(
      productId
    );

    product === undefined
  ) {
    showFeedback(

    );

    renderProducts();
    return false;
  }

  const confirmed =
    window.confirm(
      `Are you sure you want to delete ${product.name}?`
    );

  if (!confirmed) {
    showFeedback(
      `${product.name} was not deleted.`
    );

    return false;

    resetFormToCreateMode();
  }

  renderProducts();

  showFeedback(
    `${deletedProduct.name} was deleted successfully.`
  );

  return true;
}
```

## Example 1: CRUD State Functions Without the DOM

### Problem

Create, update, and delete products using modern array operations.

### Complete Code

```js
let products = [
  {
    id: 1,
    name: "Laptop",
    price: 50000,
    quantity: 2,
  },
];

let nextId = 2;

function createProduct(
  data
) {
  const newProduct = {
    id: nextId,
    ...data,
  };

  nextId++;

  products = [
    ...products,
    newProduct,
  ];

  return newProduct;
}

function updateProduct(
  id,
  updates
) {
  const exists =
    products.some(
      (product) =>
        product.id === id
    );

  if (!exists) {
    return null;
  }

  products =
    products.map(
      (product) =>
        product.id === id
          ? {
              ...product,
              ...updates,
              id: product.id,
            }
          : product
    );

  return products.find(
    (product) =>
      product.id === id
  );
}

function deleteProduct(id) {
  const product =
    products.find(
      (product) =>
        product.id === id
    );

  if (
    product === undefined
  ) {
    return null;
  }

  products =
    products.filter(
      (product) =>
        product.id !== id
    );

  return product;
}

createProduct({
  name: "Phone",
  price: 25000,
  quantity: 5,
});

updateProduct(1, {
  name: "Gaming Laptop",
  price: 65000,
});

deleteProduct(2);

console.log(products);
```

```text
]
```

# 48. Common Mistakes

## Mistake 1: State Recreated Inside an Event Handler

Incorrect:

```js
form.addEventListener(
  "submit",
  function () {
    let products = [];
  }
);
```

This creates a new local array and loses access to the real state.

---

## Mistake 2: Search Replaces Main State

Incorrect:

```js
products =
  products.filter(
    searchCondition
  );
```

Use a derived variable for temporary display filters.

---

## Mistake 3: Duplicate Validation Rejects the Edited Product

Pass:

```js
editingProductId
```

to the duplicate checker.

---

## Mistake 4: Missing `map()` Reassignment

```js
products.map(
  updateCallback
);
```

The new array is discarded.

---

## Mistake 5: Wrong Delete Condition

```js
product.id === productId
```

keeps only the selected product.

---

## Mistake 6: Missing `reduce()` Return

The accumulator becomes `undefined`.

---

## Mistake 7: Direct DOM-Only Update

A later render restores old state values.

---

## Mistake 8: State Update Without Rendering

The page remains stale.

---

## Mistake 9: Direct Listeners Added During Rendering

Repeated renders can create inconsistent event behavior.

Use one delegated parent listener.

---

## Mistake 10: User Input Inserted with `innerHTML`

Use:

```js
textContent
```

for names and other untrusted input.

# 49. Best Practices

* Keep the main state array outside event handlers.
* Treat state as the source of truth.
* Use one reusable render function.
* Re-render after successful state changes.
* Use `forEach()` for DOM rendering.
* Use `find()` when one object is needed.
* Use `findIndex()` only when a position is required.
* Use `map()` for immutable Update.
* Use `filter()` for immutable Delete and derived lists.
* Use `some()` for existence checks.
* Use `every()` for whole-state checks.
* Use `reduce()` for totals and summaries.
* Preserve stable IDs.
* Normalize strings before comparison.
* Validate before mutating state.
* Use event delegation.
* Keep rendering free from state mutation.
* Use `textContent` for user-controlled values.

## Key Takeaways

* Modern array methods can support a complete CRUD application.
* `forEach()` is appropriate for rendering side effects.
* `find()` retrieves one matching product.
* `findIndex()` retrieves a product position.
* `map()` supports immutable Update.
* `filter()` supports immutable Delete and visible-list derivation.
* `some()` prevents duplicate names.
* `every()` validates the entire state.
* `reduce()` calculates totals and summary values.
* Search and filtering should derive visible data without replacing main state.
* Stable IDs must remain separate from array positions.
* State should be updated before the DOM.
* Rendering should follow every successful state mutation.
* Event delegation works well with dynamic lists.
* The next subtopic will compare plain loops and array methods operation by operation.
