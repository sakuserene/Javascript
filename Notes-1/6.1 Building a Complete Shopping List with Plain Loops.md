# 21.9 Finding an Item by ID

```js
function findItemById(itemId) {
  for (
    let index = 0;
    index < items.length;
    index++
  ) {
    if (
      items[index].id === itemId
    ) {
      return items[index];
    }
  }

  return null;
}
```

This returns:

* The matching object.
* `null` when no item exists.

# 21.10 Finding an Item Index

```js
function findItemIndexById(
  itemId
) {
  for (
    let index = 0;
    index < items.length;
    index++
  ) {
    if (
      items[index].id === itemId
    ) {
      return index;
    }
  }

  return -1;
}
```

This is needed for deletion with `splice()`.

# 21.11 Checking Duplicate Names with a Plain Loop

```js
function itemNameExists(
  name,
  ignoredItemId = null
) {
  const normalizedName =
    name.trim().toLowerCase();

  for (
    let index = 0;
    index < items.length;
    index++
  ) {
    const item = items[index];

    if (
      item.id === ignoredItemId
    ) {
      continue;
    }

    const existingName =
      item.name
        .trim()
        .toLowerCase();

    if (
      existingName ===
      normalizedName
    ) {
      return true;
    }
  }

  return false;
}
```

During Create:

```js
itemNameExists(name);
```

During Update:

```js
itemNameExists(
  name,
  editingItemId
);
```

# 21.13 Validating the Item Name

```js
function validateItemData(data) {
  if (data.name === "") {
    return {
      isValid: false,
      message:
        "Item name is required.",
    };
  }

  if (data.name.length < 2) {
    return {
      isValid: false,
      message:
        "Item name must contain at least 2 characters.",
    };
  }

  if (data.name.length > 60) {
    return {
      isValid: false,
      message:
        "Item name cannot exceed 60 characters.",
    };
  }

  if (
    itemNameExists(
      data.name,
      editingItemId
    )
  ) {
    return {
      isValid: false,
      message:
        "An item with this name already exists.",
    };
  }

  return {
    isValid: true,
    normalizedData: {
      name: data.name,
    },
  };
}
```

# 21.15 Starting Edit Mode

function startEditingItem(
  itemId
) {

# 21.17 Updating an Item Name

```js
function updateItem(data) {
  let wasUpdated = false;

  for (
    let index = 0;
    index < items.length;
    index++
  ) {
    if (
      items[index].id ===
      editingItemId
    ) {
      items[index].name =
        data.name;

      wasUpdated = true;
      break;
    }
  }

  if (!wasUpdated) {
    showError(
      "The selected item no longer exists."
    );

    resetFormToCreateMode();
    renderItems();

    return null;
  }

  return findItemById(
    editingItemId
  );
}
```

The ID and purchased status remain unchanged.

# 21.19 Toggling Purchased Status

The purchased value is boolean:

```js
true
```

or:

```js
false
```

Toggle it:

```js
item.purchased =
  !item.purchased;
```

Complete function:

```js
function toggleItemPurchased(
  itemId
) {
  let updatedItem = null;

  for (
    let index = 0;
    index < items.length;
    index++
  ) {
    if (
      items[index].id === itemId
    ) {
      items[index].purchased =
        !items[index].purchased;

      updatedItem =
        items[index];

      break;
    }
  }

  return updatedItem;
}
```

# 21.22 Deleting an Item

```js
function deleteItemById(itemId) {
  const itemIndex =
    findItemIndexById(
      itemId
    );

  if (itemIndex === -1) {
    return null;
  }

  const removedItems =
    items.splice(
      itemIndex,
      1
    );

  return removedItems[0];
}
```

# 21.24 Complete Delete Request

```js
function handleDeleteRequest(
  itemId
) {
  clearMessages();

  const item =
    findItemById(itemId);

  if (item === null) {
    showError(
      "The selected item no longer exists."
    );

    renderItems();
    return;
  }

  const shouldDelete =
    confirmItemDeletion(item);

  if (!shouldDelete) {
    showInfo(
      `${item.name} was not deleted.`
    );

    return;
  }

  const deletedItem =
    deleteItemById(itemId);

  if (deletedItem === null) {
    showError(
      "The selected item could not be deleted."
    );

    renderItems();
    return;
  }

  if (
    editingItemId ===
    deletedItem.id
  ) {
    resetFormToCreateMode();
  }

  renderItems();

  showSuccess(
    `${deletedItem.name} was deleted successfully.`
  );
}
```

# 21.25 Counting Purchased Items with a Plain Loop

function countPurchasedItems() {

    index < items.length;

      count++;

# 21.26 Counting Unpurchased Items

function countUnpurchasedItems() {

# 21.29 Rendering One Shopping Item

```js
function renderItemCard(
  item,
  index
) {
  const card =
    document.createElement("li");

  const itemName =
    document.createElement("span");

  const itemStatus =
    document.createElement("span");

  const actions =
    document.createElement("div");

  const toggleButton =
    document.createElement("button");

  const editButton =
    document.createElement("button");

  const deleteButton =
    document.createElement("button");

  card.classList.add(
    "item-card"
  );

  card.dataset.id = item.id;

  if (item.purchased) {
    card.classList.add(
      "purchased-item"
    );
  }

  if (
    item.id === editingItemId
  ) {
    card.classList.add(
      "currently-editing"
    );
  }

  itemName.classList.add(
    "item-name"
  );

  itemName.textContent =
    `${index + 1}. ${item.name}`;

  itemStatus.textContent =
    item.purchased
      ? "Purchased"
      : "Not purchased";

  toggleButton.type = "button";

  toggleButton.textContent =
    item.purchased
      ? "Mark Unpurchased"
      : "Mark Purchased";

  toggleButton.dataset.action =
    "toggle";

  toggleButton.dataset.id =
    item.id;

  editButton.type = "button";
  editButton.textContent = "Edit";

  editButton.dataset.action =
    "edit";

  editButton.dataset.id =
    item.id;

  editButton.disabled =
    item.id === editingItemId;

  deleteButton.type = "button";
  deleteButton.textContent =
    "Delete";

  deleteButton.dataset.action =
    "delete";

  deleteButton.dataset.id =
    item.id;

  actions.append(
    toggleButton,
    editButton,
    deleteButton
  );

  card.append(
    itemName,
    itemStatus,
    actions
  );

  itemList.appendChild(card);
}
```

# 21.31 Handling Form Submission

```js
function handleItemFormSubmit(
  event
) {
  event.preventDefault();

  clearMessages();

  const rawData =
    readItemForm();

  const result =
    validateItemData(rawData);

  if (!result.isValid) {
    showError(result.message);

    itemNameInput.focus();
    return;
  }

  if (editingItemId === null) {
    const newItem =
      createItem(
        result.normalizedData
      );

    resetFormToCreateMode();
    renderItems();

    showSuccess(
      `${newItem.name} was added successfully.`
    );

    return;
  }

  const updatedItem =
    updateItem(
      result.normalizedData
    );

  if (updatedItem === null) {
    return;
  }

  const updatedName =
    updatedItem.name;

  resetFormToCreateMode();
  renderItems();

  showSuccess(
    `${updatedName} was updated successfully.`
  );
}
```

# 21.32 Handling List Clicks

```js
function handleItemListClick(
  event
) {
  const clickedButton =
    event.target.closest(
      "button"
    );

  if (clickedButton === null) {
    return;
  }

  const action =
    clickedButton.dataset.action;

  const itemId =
    Number(
      clickedButton.dataset.id
    );

  if (
    !Number.isInteger(itemId) ||
    itemId <= 0
  ) {
    showError(
      "The selected item ID is invalid."
    );

    return;
  }

  if (action === "edit") {
    startEditingItem(itemId);
    return;
  }

  if (action === "toggle") {
    clearMessages();

    const updatedItem =
      toggleItemPurchased(
        itemId
      );

    if (updatedItem === null) {
      showError(
        "The selected item no longer exists."
      );

      renderItems();
      return;
    }

    renderItems();

    const status =
      updatedItem.purchased
        ? "purchased"
        : "not purchased";

    showSuccess(
      `${updatedItem.name} is now ${status}.`
    );

    return;
  }

  if (action === "delete") {
    handleDeleteRequest(itemId);
  }
}
```

      errorMessage.textContent = "";
      successMessage.textContent = "";
      infoMessage.textContent = "";
    }

    function showError(message) {
      errorMessage.textContent =
        message;
    }

    function showSuccess(message) {
      successMessage.textContent =
        message;
    }

    function showInfo(message) {
      infoMessage.textContent =
        message;

    function generateItemId() {
      const id = nextItemId;

      nextItemId++;

      return id;
    }

    function findItemById(itemId) {
      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        if (
          items[index].id ===
          itemId
        ) {
          return items[index];
        }
      }

      return null;
    }

    function findItemIndexById(
      itemId
    ) {
      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        if (
          items[index].id ===
          itemId
        ) {
          return index;
        }
      }

      return -1;
    }

    function itemNameExists(
      name,
      ignoredItemId = null
    ) {
      const normalizedName =
        name.trim().toLowerCase();

      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        const item = items[index];

        if (
          item.id ===
          ignoredItemId
        ) {
          continue;
        }

        if (
          item.name
            .trim()
            .toLowerCase() ===
          normalizedName
        ) {
          return true;
        }
      }

      return false;
    }

    function readItemForm() {
      return {
        name:
          itemNameInput.value.trim(),
      };
    }

    function validateItemData(
      data
    ) {
      if (data.name === "") {
        return {
          isValid: false,
          message:
            "Item name is required.",
        };
      }

      if (data.name.length < 2) {
        return {
          isValid: false,
          message:
            "Item name must contain at least 2 characters.",
        };
      }

      if (data.name.length > 60) {
        return {
          isValid: false,
          message:
            "Item name cannot exceed 60 characters.",
        };
      }

      if (
        itemNameExists(
          data.name,
          editingItemId
        )
      ) {
        return {
          isValid: false,
          message:
            "An item with this name already exists.",
        };
      }

      return {
        isValid: true,
        normalizedData: {
          name: data.name,
        },
      };
    }

    function createItem(data) {
      const newItem = {
        id: generateItemId(),
        name: data.name,
        purchased: false,
      };

      items.push(newItem);

      return newItem;
    }

    function updateItem(data) {
      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        if (
          items[index].id ===
          editingItemId
        ) {
          items[index].name =
            data.name;

          return items[index];
        }
      }

      return null;
    }

    function toggleItemPurchased(
      itemId
    ) {
      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        if (
          items[index].id ===
          itemId
        ) {
          items[index].purchased =
            !items[index].purchased;

          return items[index];
        }
      }

      return null;
    }

    function deleteItemById(
      itemId
    ) {
      const itemIndex =
        findItemIndexById(
          itemId
        );

      if (itemIndex === -1) {
        return null;
      }

      const removedItems =
        items.splice(
          itemIndex,
          1
        );

      return removedItems[0];
    }

    function confirmItemDeletion(
      item
    ) {
      return window.confirm(
        `Are you sure you want to delete ${item.name}?`
      );
    }

    function countPurchasedItems() {
      let count = 0;

      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        if (items[index].purchased) {
          count++;
        }
      }

      return count;
    }

    function countUnpurchasedItems() {
      let count = 0;

      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        if (!items[index].purchased) {
          count++;
        }
      }

      return count;
    }

    function renderSummary() {
      totalCount.textContent =
        `Total items: ${items.length}`;

      purchasedCount.textContent =
        `Purchased: ${countPurchasedItems()}`;

      unpurchasedCount.textContent =
        `Not purchased: ${countUnpurchasedItems()}`;
    }

    function renderEmptyState() {
      const emptyState =
        document.createElement("li");

      emptyState.classList.add(
        "empty-state"
      );

      emptyState.textContent =
        "No shopping items available.";

      itemList.appendChild(
        emptyState
      );
    }

    function renderItemCard(
      item,
      index
    ) {
      const card =
        document.createElement("li");

      const itemName =
        document.createElement("strong");

      const itemStatus =
        document.createElement("span");

      const actions =
        document.createElement("div");

      const toggleButton =
        document.createElement("button");

      const editButton =
        document.createElement("button");

      const deleteButton =
        document.createElement("button");

      card.classList.add(
        "item-card"
      );

      card.dataset.id = item.id;

      if (item.purchased) {
        card.classList.add(
          "purchased-item"
        );
      }

      if (
        item.id === editingItemId
      ) {
        card.classList.add(
          "currently-editing"
        );
      }

      itemName.classList.add(
        "item-name"
      );

      itemName.textContent =
        `${index + 1}. ${item.name}`;

      itemStatus.textContent =
        item.purchased
          ? "Purchased"
          : "Not purchased";

      actions.classList.add(
        "item-actions"
      );

      toggleButton.type =
        "button";

      toggleButton.textContent =
        item.purchased
          ? "Mark Unpurchased"
          : "Mark Purchased";

      toggleButton.dataset.action =
        "toggle";

      toggleButton.dataset.id =
        item.id;

      editButton.type = "button";
      editButton.textContent =
        "Edit";

      editButton.dataset.action =
        "edit";

      editButton.dataset.id =
        item.id;

      editButton.disabled =
        item.id === editingItemId;

      deleteButton.type =

      deleteButton.textContent =
        "Delete";

      deleteButton.dataset.action =
        "delete";

      deleteButton.dataset.id =
        item.id;

        editButton,
        deleteButton

      card.append(

        itemStatus,
        actions
      );

      itemList.appendChild(card);

    function renderItems() {
      itemList.innerHTML = "";

      renderSummary();

      if (items.length === 0) {
        renderEmptyState();
        return;
      }

      for (
        let index = 0;
        index < items.length;
        index++
      ) {
        renderItemCard(
          items[index],
          index
        );
      }
    }

    function resetFormToCreateMode() {
      editingItemId = null;

      itemForm.reset();

      submitButton.textContent =
        "Add Item";

      cancelEditButton.hidden =
        true;

      itemNameInput.focus();
    }

    function startEditingItem(
      itemId
    ) {
      clearMessages();

      const item =
        findItemById(itemId);

      if (item === null) {
        showError(
          "The selected item no longer exists."
        );

        renderItems();
        return;
      }

      editingItemId = item.id;

      itemNameInput.value =
        item.name;

      submitButton.textContent =
        "Update Item";

      cancelEditButton.hidden =
        false;

      itemNameInput.focus();
      itemNameInput.select();

      renderItems();
    }

    function handleDeleteRequest(
      itemId
    ) {
      clearMessages();

      const item =
        findItemById(itemId);

      if (item === null) {
        showError(
          "The selected item no longer exists."
        );

        renderItems();
        return;
      }

      const shouldDelete =
        confirmItemDeletion(item);

      if (!shouldDelete) {
        showInfo(
          `${item.name} was not deleted.`
        );

        return;
      }

      const deletedItem =
        deleteItemById(itemId);

      if (deletedItem === null) {
        showError(
          "The selected item could not be deleted."
        );

        renderItems();
        return;
      }

      if (
        editingItemId ===
        deletedItem.id
      ) {
        resetFormToCreateMode();
      }

      renderItems();

      showSuccess(
        `${deletedItem.name} was deleted successfully.`
      );
    }

    function handleItemFormSubmit(
      event
    ) {
      event.preventDefault();

      clearMessages();

      const rawData =
        readItemForm();

      const result =
        validateItemData(rawData);

      if (!result.isValid) {
        showError(result.message);

        itemNameInput.focus();
        return;
      }

      if (
        editingItemId === null
      ) {
        const newItem =
          createItem(
            result.normalizedData
          );

        resetFormToCreateMode();
        renderItems();

        showSuccess(
          `${newItem.name} was added successfully.`
        );

        return;
      }

      const updatedItem =
        updateItem(
          result.normalizedData
        );

      if (updatedItem === null) {
        showError(
          "The selected item no longer exists."
        );

        resetFormToCreateMode();
        renderItems();

        return;
      }

      const updatedName =
        updatedItem.name;

      resetFormToCreateMode();
      renderItems();

      showSuccess(
        `${updatedName} was updated successfully.`
      );
    }

    function handleItemListClick(
      event
    ) {
      const clickedButton =
        event.target.closest(
          "button"
        );

      if (clickedButton === null) {
        return;
      }

      const action =
        clickedButton.dataset.action;

      const itemId =
        Number(
          clickedButton.dataset.id
        );

      if (
        !Number.isInteger(itemId) ||
        itemId <= 0
      ) {
        showError(
          "The selected item ID is invalid."
        );

        return;
      }

      if (action === "edit") {
        startEditingItem(itemId);
        return;
      }

      if (action === "toggle") {
        clearMessages();

        const updatedItem =
          toggleItemPurchased(
            itemId
          );

        if (
          updatedItem === null
        ) {
          showError(
            "The selected item no longer exists."
          );

          renderItems();
          return;
        }

        renderItems();

        const newStatus =
          updatedItem.purchased
            ? "purchased"
            : "not purchased";

        showSuccess(
          `${updatedItem.name} is now ${newStatus}.`
        );

        return;
      }

      if (action === "delete") {
        handleDeleteRequest(
          itemId
        );
      }
    }

    itemForm.addEventListener(
      "submit",
      handleItemFormSubmit
    );

    itemList.addEventListener(
      "click",
      handleItemListClick
    );

    cancelEditButton.addEventListener(
      "click",
      function () {
        clearMessages();

        resetFormToCreateMode();
        renderItems();

        showInfo(
          "Editing was canceled."
        );
      }
    );

    resetFormToCreateMode();
    renderItems();
  </script>

# 21.44 Common Beginner Mistakes

## Mistake 1: Recreating the State Inside the Submit Handler

```js
function handleSubmit() {
  let items = [];
}
```

Every submission loses previous items.

---

## Mistake 2: Creating a New Item During Edit Mode

The submit handler must check:

```js
editingItemId === null
```

---

## Mistake 3: Generating a New ID During Update

The original ID must remain unchanged.

---

## Mistake 4: Duplicate Checking Rejects the Current Item

Ignore:

```js
editingItemId
```

during edit validation.

---

## Mistake 5: Toggling Only the DOM

Changing button text without changing the object causes state inconsistency.

---

## Mistake 6: Deleting Only the DOM Card

The item returns during the next render.

---

## Mistake 7: Passing an ID Directly to `splice()`

Find its current index first.

---

## Mistake 8: Forgetting to Re-render

State changes but the page remains stale.

---

## Mistake 9: Adding Listeners During Every Render

This can produce duplicated behavior.

Use parent event delegation.

---

## Mistake 10: Using `innerHTML` for User Input

Use:

```js
textContent
```

for shopping-item names.

  items[index].purchased =
    !items[index].purchased;

## Key Takeaways

* A complete CRUD application can be built using plain JavaScript loops.
* The shopping-list array is the main source of truth.
* Create uses validated form input and `push()`.
* Read uses a reusable render function and a standard `for` loop.
* Update includes editing names and toggling purchased status.
* Delete uses a stable ID, a current index, confirmation, and `splice()`.
* The DOM must be re-rendered after every successful state mutation.
* Summary counts should be derived from the current array.
* Event delegation supports dynamically rendered buttons.
* Create mode and edit mode can share one form.
* Missing and stale items must be handled safely.
